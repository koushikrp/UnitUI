Slider

Default slider

Default slider can be used with price-range class

Price range

0

4000

							
								<div class="pt-32">

									<h5 class="h6">Price range</h5>

									<input
									  type="range"
									  class="price-range"
									  min="0"
									  max="4000"
									  value="1000"
									  step="500"
									  />

									<span class="selected-price"></span>

								  </div>

JS:

							
								<script>

									const slider = document.querySelector('.price-range');

									const selectedPrice = document.querySelector('.selected-price');

									selectedPrice.innerHTML = slider.value;

									console.log(slider);

									console.log(selectedPrice);

									slider.oninput = () => (selectedPrice.innerHTML = slider.value);

								  </script>
